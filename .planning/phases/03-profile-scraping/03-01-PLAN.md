    **src/scrape/scraper.py**:
    - Class BlockDetectedError(Exception): raised when scraping is blocked
    - Class ScrapeError(Exception): raised after max retries exhausted
    - Class XProfileScraper:
      - __init__(self, cache_dir: Path, min_delay: float = 2.0, max_delay: float = 5.0, max_attempts: int = 3)
        - Store cache_dir, min_delay, max_delay, max_attempts
        - Parse robots.txt using urllib.robotparser.RobotFileParser at https://x.com/robots.txt
        - Store crawl_delay from robots (default 1.0 if not found)
        - Create self._session = curl_cffi.requests.Session(impersonate="chrome")
        - Create self._backoff = ExponentialBackoff(base=2.0, max_delay=300.0)
      - scrape_profile(self, username: str) -> dict | None:
        - URL: f"https://x.com/{username}"
        - For each attempt in range(max_attempts):
          - response = self._session.get(url, timeout=15)
          - If status == 429: sleep self._backoff.delay(), backoff._attempt += 1, continue
          - If status != 200: log warning, return None
          - If is_blocked(response): log warning, return None
          - Parse with BeautifulSoup(response.text, "lxml")
          - Call parse_profile_fields(soup) to extract fields
          - Update self._cache_dir / f"{username}.json" with scraped fields
          - Return scraped_fields dict
        - After max_attempts: log error, return None (graceful degradation per SCRAPE-05)
      - is_blocked(self, response) -> bool:
        - Returns True if any of: status == 429, or (status == 200 and not response.text.strip()), or ("challenges" in response.url), or (status == 200 and "<title>" not in response.text)
      - _apply_delay(self) -> None:
        - Compute random.uniform(min_delay, max_delay) + jitter (random.uniform(0, 0.5))
        - Also honor crawl_delay from robots.txt as minimum floor
        - time.sleep(calculated_delay)
      - _cache_scraped_fields(self, username: str, scraped: dict) -> None:
        - Read existing cache file from self._cache_dir / f"{username}.json" (or create minimal dict if not exists)
        - Add all scraped fields to the dict
        - Write back to same file with json.dumps(..., indent=2)

    **src/scrape/__init__.py** — add scrape_all() function before the ROBOTS_TXT_LEGAL string constant:
    ```python
    from dataclasses import dataclass
    from pathlib import Path
    import json, logging, glob
    from src.scrape.scraper import XProfileScraper

    logger = logging.getLogger(__name__)

    @dataclass
    class ScrapeResult:
        total: int
        scraped: int
        skipped: int  # needs_scraping was False
        failed: int   # exceptions
        blocked: int  # blocked responses

    def scrape_all(
        cache_dir: str | Path = Path("data/enrichment"),
        min_delay: float = 2.0,
        max_delay: float = 5.0,
    ) -> ScrapeResult:
        """Scrape supplemental profile fields for all accounts with needs_scraping=True.

        Args:
            cache_dir: Directory containing {account_id}.json cache files from Phase 2.
            min_delay: Minimum seconds between requests (default 2.0).
            max_delay: Maximum seconds between requests (default 5.0).

        Returns:
            ScrapeResult with counts.
        """
        cache_dir = Path(cache_dir)
        scraper = XProfileScraper(cache_dir=cache_dir, min_delay=min_delay, max_delay=max_delay)

        total = 0
        scraped = 0
        skipped = 0
        failed = 0
        blocked = 0

        # Find all account cache files
        cache_files = sorted(cache_dir.glob("*.json"))
        # Exclude special files
        account_files = [f for f in cache_files if f.stem not in ("suspended", "protected", "errors")]

        for cache_path in account_files:
            try:
                with open(cache_path, encoding="utf-8") as f:
                    account = json.load(f)
            except (json.JSONDecodeError, OSError) as e:
                logger.warning("Could not read %s: %s", cache_path, e)
                failed += 1
                continue

            account_id = cache_path.stem
            total += 1

            # D-06: Skip if already fully populated (no needs_scraping flag or False)
            if not account.get("needs_scraping"):
                skipped += 1
                continue

            # Also skip if already has professional_category (already scraped)
            if account.get("professional_category"):
                skipped += 1
                continue

            # Scrape the profile
            result = scraper.scrape_profile(account.get("username", account_id))
            if result is None:
                # Blocked or failed — graceful degradation, continue
                if scraper._last_blocked:
                    blocked += 1
                else:
                    failed += 1
                continue

            # result already merged into cache by scraper._cache_scraped_fields
            scraped += 1
            logger.info("Scraped %s (%d/%d)", account_id, scraped, len(account_files))

        logger.info(
            "Scraping complete: %d/%d scraped, %d skipped, %d failed, %d blocked",
            scraped, total, skipped, failed, blocked,
        )
        return ScrapeResult(
            total=total,
            scraped=scraped,
            skipped=skipped,
            failed=failed,
            blocked=blocked,
        )
    ```
