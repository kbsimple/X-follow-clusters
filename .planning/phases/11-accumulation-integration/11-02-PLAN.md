class TestIntegrationAccumulation:
    """End-to-end integration tests for tweet accumulation."""

    def test_first_fetch_persists_tweets_and_updates_account_json(
        self, temp_tweet_cache: TweetCache, tmp_path: Path
    ) -> None:
        """First enrichment run: empty cache -> API fetch -> persist -> account JSON update."""
        from src.enrich.api_client import XEnrichmentClient
        from unittest.mock import MagicMock, patch

        user_id = "first_fetch_user"

        # Create account JSON file
        account_json = tmp_path / "enrichment" / f"{user_id}.json"
        account_json.parent.mkdir(parents=True, exist_ok=True)
        account_json.write_text(json.dumps({"id": user_id, "username": "test_user"}))

        # Mock API to return 3 tweets
        new_tweets = [
            {"id": "tweet_1", "text": "First tweet", "created_at": "2024-05-21T10:00:00.000Z",
             "public_metrics": {"like_count": 1}},
            {"id": "tweet_2", "text": "Second tweet", "created_at": "2024-05-21T11:00:00.000Z",
             "public_metrics": {"like_count": 2}},
            {"id": "tweet_3", "text": "Third tweet", "created_at": "2024-05-21T12:00:00.000Z",
             "public_metrics": {"like_count": 3}},
        ]

        with patch.object(XEnrichmentClient, '_fetch_tweets_from_api') as mock_fetch:
            mock_fetch.return_value = new_tweets
            auth = MagicMock()
            auth.access_token = "test_token"
            client = XEnrichmentClient(auth=auth, cache_dir=tmp_path / "enrichment")

            result = client.get_recent_tweets(user_id, max_tweets=50, tweet_cache=temp_tweet_cache)

        # Verify cache has 3 tweets
        cached = temp_tweet_cache.load_tweets(user_id)
        assert cached.count == 3

        # Verify newest tweet is watermark
        watermark = temp_tweet_cache.get_newest_tweet_id(user_id)
        assert watermark == "tweet_3"  # Highest created_at

    def test_subsequent_fetch_uses_watermark_and_merges(
        self, temp_tweet_cache: TweetCache, tmp_path: Path
    ) -> None:
        """Subsequent run: cache exists -> since_id used -> new + cached merged."""
        from src.enrich.api_client import XEnrichmentClient
        from unittest.mock import MagicMock, patch

        user_id = "subsequent_fetch_user"

        # Pre-populate cache with 3 old tweets
        old_tweets = [
            {"id": "old_1", "text": "Old tweet 1", "created_at": "2024-05-20T10:00:00.000Z",
             "public_metrics": {"like_count": 1}},
            {"id": "old_2", "text": "Old tweet 2", "created_at": "2024-05-20T11:00:00.000Z",
             "public_metrics": {"like_count": 2}},
            {"id": "old_3", "text": "Old tweet 3", "created_at": "2024-05-20T12:00:00.000Z",
             "public_metrics": {"like_count": 3}},
        ]
        temp_tweet_cache.persist_tweets(user_id, old_tweets)

        # Get watermark (should be old_3)
        watermark = temp_tweet_cache.get_newest_tweet_id(user_id)
        assert watermark == "old_3"

        # Mock API to return 2 new tweets
        new_tweets = [
            {"id": "new_1", "text": "New tweet 1", "created_at": "2024-05-21T10:00:00.000Z",
             "public_metrics": {"like_count": 10}},
            {"id": "new_2", "text": "New tweet 2", "created_at": "2024-05-21T11:00:00.000Z",
             "public_metrics": {"like_count": 20}},
        ]

        with patch.object(XEnrichmentClient, '_fetch_tweets_from_api') as mock_fetch:
            mock_fetch.return_value = new_tweets
            auth = MagicMock()
            auth.access_token = "test_token"
            client = XEnrichmentClient(auth=auth)

            result = client.get_recent_tweets(user_id, max_tweets=50, tweet_cache=temp_tweet_cache)

            # Verify since_id was passed to API
            call_kwargs = mock_fetch.call_args[1]
            assert call_kwargs.get('since_id') == watermark

        # Verify merged result has 5 tweets
        assert len(result) == 5

        # Verify new tweets come first (most recent)
        assert result[0]["id"] == "new_2"
        assert result[1]["id"] == "new_1"

        # Verify cache now has 5 tweets
        cached = temp_tweet_cache.load_tweets(user_id)
        assert cached.count == 5

    def test_deduplication_same_tweet_id_stored_once(
        self, temp_tweet_cache: TweetCache
    ) -> None:
        """Same tweet ID from API stored only once via PRIMARY KEY constraint."""
        from src.enrich.api_client import XEnrichmentClient
        from unittest.mock import MagicMock, patch

        user_id = "dedupe_user"

        # Mock API to return same tweet twice
        duplicate_tweets = [
            {"id": "same_tweet", "text": "Same tweet", "created_at": "2024-05-21T10:00:00.000Z"},
            {"id": "same_tweet", "text": "Same tweet", "created_at": "2024-05-21T10:00:00.000Z"},
        ]

        with patch.object(XEnrichmentClient, '_fetch_tweets_from_api') as mock_fetch:
            mock_fetch.return_value = duplicate_tweets
            auth = MagicMock()
            auth.access_token = "test_token"
            client = XEnrichmentClient(auth=auth)

            result = client.get_recent_tweets(user_id, max_tweets=50, tweet_cache=temp_tweet_cache)

        # Verify cache has only 1 tweet
        cached = temp_tweet_cache.load_tweets(user_id)
        assert cached.count == 1

    def test_accumulation_across_multiple_runs(
        self, temp_tweet_cache: TweetCache
    ) -> None:
        """Tweet count grows across multiple enrichment runs."""
        from src.enrich.api_client import XEnrichmentClient
        from unittest.mock import MagicMock, patch

        user_id = "multi_run_user"

        # Run 1: fetch 3 tweets
        run1_tweets = [
            {"id": f"run1_{i}", "text": f"Run 1 tweet {i}", "created_at": f"2024-05-20T1{i}:00:00.000Z"}
            for i in range(3)
        ]

        with patch.object(XEnrichmentClient, '_fetch_tweets_from_api') as mock_fetch:
            mock_fetch.return_value = run1_tweets
            auth = MagicMock()
            auth.access_token = "test_token"
            client = XEnrichmentClient(auth=auth)
            client.get_recent_tweets(user_id, max_tweets=50, tweet_cache=temp_tweet_cache)

        cached1 = temp_tweet_cache.load_tweets(user_id)
        assert cached1.count == 3

        # Run 2: fetch 2 more tweets
        run2_tweets = [
            {"id": f"run2_{i}", "text": f"Run 2 tweet {i}", "created_at": f"2024-05-21T1{i}:00:00.000Z"}
            for i in range(2)
        ]

        with patch.object(XEnrichmentClient, '_fetch_tweets_from_api') as mock_fetch:
            mock_fetch.return_value = run2_tweets
            client.get_recent_tweets(user_id, max_tweets=50, tweet_cache=temp_tweet_cache)

        cached2 = temp_tweet_cache.load_tweets(user_id)
        assert cached2.count == 5  # 3 + 2 = 5 total

        # Run 3: API returns empty (no new tweets)
        with patch.object(XEnrichmentClient, '_fetch_tweets_from_api') as mock_fetch:
            mock_fetch.return_value = []
            client.get_recent_tweets(user_id, max_tweets=50, tweet_cache=temp_tweet_cache)

        cached3 = temp_tweet_cache.load_tweets(user_id)
        assert cached3.count == 5  # Still 5 (historical preserved)

    def test_watermark_tracks_newest_tweet(
        self, temp_tweet_cache: TweetCache

            {"id": "mid_tweet", "text": "Mid", "created_at": "2024-05-20T10:00:00.000Z"},
        ]
        temp_tweet_cache.persist_tweets(user_id, tweets)

        watermark = temp_tweet_cache.get_newest_tweet_id(user_id)
        assert watermark == "new_tweet"  # Highest created_at

        # Mock API to return tweets

        with patch.object(XEnrichmentClient, '_fetch_tweets_from_api') as mock_fetch:
