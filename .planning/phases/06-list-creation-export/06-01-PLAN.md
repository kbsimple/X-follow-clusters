    Function signature and behavior:
    ```python
    def create_lists_from_clusters(
        approved_clusters: list[dict],
        client,
        dry_run: bool = False
    ) -> dict:
        """
        Orchestrates the full list creation flow for approved clusters.

        Args:
            approved_clusters: List of approved cluster dicts from registry
            client: Authenticated tweepy.Client instance
            dry_run: If True, only simulate without making API calls

        Returns:
            dict with keys: "created" (list of created list ids),
                            "skipped" (list of skipped cluster names),
                            "errors" (list of error messages)
        """
        results = {"created": [], "skipped": [], "errors": []}

        # 1. Verify credentials
        verify_credentials_before_listCreation()

        # 2. Run list_size_validation to filter clusters by 5-50 size and account limit
        valid_clusters = []
        for cluster in approved_clusters:
            if cluster["size"] < 5 or cluster["size"] > 50:
                results["skipped"].append(
                    f"{cluster['cluster_name']} (size {cluster['size']} out of range)"
                )
            else:
                valid_clusters.append(cluster)

        # 3. Pre-check conflicts with existing lists
        conflicts = precheck_conflicts(client, valid_clusters)
        if conflicts:
            # Return conflicts for CLI to handle via questionary prompts
            return {"created": [], "skipped": valid_clusters, "errors": [], "conflicts": list(conflicts.keys())}

        # 4. Create lists (or simulate in dry-run mode)
        for cluster in valid_clusters:
            try:
                if dry_run:
                    logging.info(f"[DRY-RUN] Would create list: {cluster['cluster_name']} with {cluster['size']} members")
                else:
                    list_id = create_list_from_cluster(client, cluster)
                    usernames = [m["username"] for m in cluster["members"]]
                    add_members_chunked(client, list_id, usernames)
                    results["created"].append(list_id)
                    logging.info(f"Created list '{cluster['cluster_name']}' with {len(usernames)} members")
                    time.sleep(0.5)  # Rate limit between list creations
            except ListCreationError as e:
                results["errors"].append(f"{cluster['cluster_name']}: {e}")
            except Exception as e:
                results["errors"].append(f"{cluster['cluster_name']}: {e}")

        return results
    ```

    The function handles:
    - Empty approved_clusters list gracefully (returns empty results)
    - Clusters skipped by size validation (added to "skipped")
    - Errors from create_list_from_cluster (caught, added to "errors", continues with remaining clusters)
    - list_size_validation runs before precheck_conflicts
